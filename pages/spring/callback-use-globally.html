Triangle.java

public class Triangle implements {

      

       private Point pointA;

       private Point pointB;

       private Point pointC;

      

       public Point getPointA() {

              return pointA;

       }

 

       public void setPointA(Point pointA) {

              this.pointA = pointA;

       }

 

       public Point getPointB() {

              return pointB;

       }

      

       public void setPointB(Point pointB) {

              this.pointB = pointB;

       }

 

       public Point getPointC() {

              return pointC;

       }

      

       public void setPointC(Point pointC) {

              this.pointC = pointC;

       }

 

       public void draw() {

              System.out.println("Triangle Draw ,"

                           + " Point A "+getPointA().getPointX()+","+getPointA().getPointY()

                           + ", Pint B "+getPointB().getPointX()+", "+getPointB().getPointY()

                           + ", Point C "+getPointC().getPointX()+", "+ getPointC().getPointY());

       }

 

      

      

       public void myInit() {

              System.out.println("Bean Initializing ..Init method called for Triangle");

       }

      

      

       public void myDestory() {

              System.out.println("Destroy method called for Triangle");

       }

}

bean.xml

<?xml version="1.0" encoding="UTF-8"?>

<beans xmlns="http://www.springframework.org/schema/beans"

       xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"

       xsi:schemaLocation="http://www.springframework.org/schema/beans http://www.springframework.org/schema/beans/spring-beans.xsd">

 

<bean id="triangle" class="com.anu.tqa.lifecyclecallbacks.Triangle" init-method="myInit" destroy-method="myDestory">

       <property name="pointA" ref="pint0"/>

       <property name="pointB" ref="pint1"/>

       <property name="pointC" ref="pint2"/>

</bean>

 

<bean id="pint0" class="com.anu.tqa.lifecyclecallbacks.Point">

       <property name="pointX" value="0"/>

       <property name="pointY" value="0"/>

</bean>

 

<bean id="pint1" class="com.anu.tqa.lifecyclecallbacks.Point">

       <property name="pointX" value="0"/>

       <property name="pointY" value="20"/>

</bean>

 

<bean id="pint2" class="com.anu.tqa.lifecyclecallbacks.Point">

       <property name="pointX" value="20"/>

</bean>
