Binary Literals with prefix "0b"

  1.  Bean Definition Inheritance

In multiple beans if we have some common bean, then we can use that common bean using the bean definition inheritance. We can inherit that common bean across other bean. Now the parent bean which has all the common definition can be a bean itself or can be an abstract template.

16.1 MainClass.java

public class MainClass {

       public static void main(String[] args) {

              ApplicationContext context = new ClassPathXmlApplicationContext("bean.xml");

              Triangle triangle = (Triangle) context.getBean("triangle1");

              triangle.draw();

       }

}

 

16.2 bean.xml

 

<?xml version="1.0" encoding="UTF-8"?>

<beans xmlns="http://www.springframework.org/schema/beans"

       xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"

       xsi:schemaLocation="http://www.springframework.org/schema/beans http://www.springframework.org/schema/beans/spring-beans.xsd">

 

<bean id="parentTriangle" class="com.anu.tqa.beandefinitioninheritance.Triangle">

       <property name="pointA" ref="pointA"/>

</bean>

 

<bean id="triangle1" class="com.anu.tqa.beandefinitioninheritance.Triangle" parent="parentTriangle">

       <property name="pointB" ref="pointB"/>

       <property name="pointC" ref="pointC"/>

</bean>

 

<bean id="triangle2" class="com.anu.tqa.beandefinitioninheritance.Triangle" parent="parentTriangle">

       <property name="pointC" ref="pointC"/>

</bean>

 

<bean id="pointA" class="com.anu.tqa.beandefinitioninheritance.Point">

       <property name="pointX" value="0"/>

       <property name="pointY" value="0"/>

</bean>

 

<bean id="pointB" class="com.anu.tqa.beandefinitioninheritance.Point">

       <property name="pointX" value="0"/>

       <property name="pointY" value="20"/>

</bean>

 

<bean id="pointC" class="com.anu.tqa.beandefinitioninheritance.Point">

       <property name="pointX" value="20"/>

       <property name="pointY" value="0"/>

</bean>

</beans>

16.3 List (Collection)

For the case of List, the parent will be override from child, but we can merge the list of parent and child.

16.3.1 Triangle.java

public class TriangleList {

      

       private List<Point> points;

      

       public List<Point> getPoints() {

              return points;

       }

 

 

       public void setPoints(List<Point> points) {

              this.points = points;

       }

 

       public void draw() {

              for(Point pointObj : points) {

                     System.out.println("Triangle Draw ,"

                                  + " Point "+pointObj.getPointX()+","+pointObj.getPointY());

              }     

       }

}

16.3.2 bean.xml

<?xml version="1.0" encoding="UTF-8"?>

<beans xmlns="http://www.springframework.org/schema/beans"

       xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"

       xsi:schemaLocation="http://www.springframework.org/schema/beans http://www.springframework.org/schema/beans/spring-beans.xsd">

 

<bean id="parentTriangle" class="com.anu.tqa.beandefinitioninheritance.Triangle">

       <property name="points">

              <list>

                     <ref bean="pointA"/>

              </list>

       </property>

</bean>

 

<bean id="triangle1" class="com.anu.tqa.beandefinitioninheritance.Triangle" parent="parentTriangle">

       <property name="points">

       <!--  this will merge collection pointA with pointB -->

              <list merge="true">

                     <ref bean="pointB"/>

              </list>

       </property>

</bean>

 

<bean id="triangle2" class="com.anu.tqa.beandefinitioninheritance.Triangle" parent="parentTriangle">

       <property name="pointC" ref="pointC"/>

</bean>

 

<bean id="pointA" class="com.anu.tqa.beandefinitioninheritance.Point">

       <property name="pointX" value="0"/>

       <property name="pointY" value="0"/>

</bean>

 

<bean id="pointB" class="com.anu.tqa.beandefinitioninheritance.Point">

       <property name="pointX" value="0"/>

       <property name="pointY" value="20"/>

</bean>

 

<bean id="pointC" class="com.anu.tqa.beandefinitioninheritance.Point">

       <property name="pointX" value="20"/>

       <property name="pointY" value="0"/>

</bean>

</beans>

 

16.4 Abstract Bean

It can be possible to declare the bean as abstract. That bean will not be initializing, it will be act as a template

<?xml version="1.0" encoding="UTF-8"?>

<beans xmlns="http://www.springframework.org/schema/beans"

       xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"

       xsi:schemaLocation="http://www.springframework.org/schema/beans http://www.springframework.org/schema/beans/spring-beans.xsd">

 

<bean id="parentTriangle" class="com.anu.tqa.beandefinitioninheritance.Triangle">

       <property name="points">

              <list>

                     <ref bean="pointA"/>

              </list>

       </property>

</bean>

 

<bean id="triangle1" class="com.anu.tqa.beandefinitioninheritance.Triangle" parent="parentTriangle" abstract="true">

       <property name="points">

       <!--  this will merge collection pointA with pointB -->

              <list merge="true">

                     <ref bean="pointB"/>

              </list>

       </property>

</bean>

 

<bean id="triangle2" class="com.anu.tqa.beandefinitioninheritance.Triangle" parent="parentTriangle">

       <property name="pointC" ref="pointC"/>

</bean>

 

<bean id="pointA" class="com.anu.tqa.beandefinitioninheritance.Point">

       <property name="pointX" value="0"/>

       <property name="pointY" value="0"/>

</bean>

 

<bean id="pointB" class="com.anu.tqa.beandefinitioninheritance.Point">

</bean>

 

<bean id="pointC" class="com.anu.tqa.beandefinitioninheritance.Point">
