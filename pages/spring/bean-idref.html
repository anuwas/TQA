IDREF

“ref” is using for reference . a ref is using to refer the name or id of any bean. To restrict the reference only for id “idref” is using. It is one kind of xml validation in spring. So that the reference only working for id not name.

bean.xml

<?xml version="1.0" encoding="UTF-8"?>

<beans xmlns="http://www.springframework.org/schema/beans"

       xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"

       xsi:schemaLocation="http://www.springframework.org/schema/beans http://www.springframework.org/schema/beans/spring-beans.xsd">

 

<bean id="triangle" class="com.anu.tqa.idref.Triangle">

       <property name="pointA" >

              <idref bean="pint0"/>

       </property>

       <property name="pointB" >

       <!-- id is not required inner bean , id required only outside bean -->

              <bean  class="com.anu.tqa.idref.Point">

                     <property name="pointX" value="0"/>

                     <property name="pointY" value="20"/>

              </bean>

       </property>

       <property name="pointC" >

       <!-- id is not required inner bean , id required only outside bean -->

              <bean  class="com.anu.tqa.idref.Point">

                     <property name="pointX" value="20"/>

                     <property name="pointY" value="0"/>

              </bean>

       </property>

</bean>

 

<bean id="pint0" class="com.anu.tqa.idref.Point">

       <property name="pointX" value="0"/>

       <property name="pointY" value="0"/>

</bean>

 

</beans>
